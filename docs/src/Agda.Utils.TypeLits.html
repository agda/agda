  SFalse :: SBool 'False

-- | A known boolean is one we can obtain a singleton for.
--   Concrete values are trivially known.

class KnownBool (b :: Bool) where
  boolSing :: SBool b

instance KnownBool 'True where
  boolSing :: SBool 'True

instance KnownBool 'False where
  boolSing :: SBool 'False
